---
title: 'Get OpenAI Status'
api: 'GET /openai/status'
description: 'Check OpenAI service connectivity and status'
---

## Overview

Checks the connectivity and status of OpenAI services integrated with the Judge API. This endpoint helps diagnose integration issues and monitor service availability.

## Response

<ResponseField name="status" type="string" required>
  Overall OpenAI service status
  - `operational` - All services working normally
  - `degraded` - Some services experiencing issues
  - `outage` - Major service disruption
  - `maintenance` - Scheduled maintenance
</ResponseField>

<ResponseField name="services" type="object" required>
  Status of individual OpenAI services
  
  <Expandable title="service properties">
    <ResponseField name="completions_api" type="object" required>
      Text completion service status
      
      <Expandable title="completions properties">
        <ResponseField name="status" type="string" required>
          Service status (operational, degraded, down)
        </ResponseField>
        
        <ResponseField name="response_time_ms" type="number">
          Average response time in milliseconds
        </ResponseField>
        
        <ResponseField name="last_checked" type="datetime" required>
          Last status check timestamp
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="embeddings_api" type="object">
      Embeddings service status (if used)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="rate_limits" type="object" required>
  Current rate limit status
  
  <Expandable title="rate limit properties">
    <ResponseField name="requests_remaining" type="integer" required>
      Requests remaining in current window
    </ResponseField>
    
    <ResponseField name="tokens_remaining" type="integer" required>
      Tokens remaining in current window
    </ResponseField>
    
    <ResponseField name="reset_time" type="datetime" required>
      When limits reset
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="usage_stats" type="object" required>
  Recent usage statistics
  
  <Expandable title="usage properties">
    <ResponseField name="requests_last_hour" type="integer" required>
      Requests made in the last hour
    </ResponseField>
    
    <ResponseField name="tokens_used_today" type="integer" required>
      Tokens consumed today
    </ResponseField>
    
    <ResponseField name="average_response_time" type="number" required>
      Average response time over last 24 hours (ms)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="health_indicators" type="array">
  Health indicators and potential issues
  
  <Expandable title="indicator properties">
    <ResponseField name="indicator" type="string" required>
      Health indicator name
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Indicator status (healthy, warning, critical)
    </ResponseField>
    
    <ResponseField name="message" type="string">
      Description or recommendation
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="last_updated" type="datetime" required>
  When status information was last updated
</ResponseField>
