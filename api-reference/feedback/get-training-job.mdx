---
title: 'Get Training Job'
api: 'GET /training/jobs/{job_id}'
description: 'Retrieve training job status and results'
---

## Overview

Retrieves detailed information about a training job, including current status, progress metrics, training results, and performance improvements achieved.

## Path Parameters

<ParamField path="job_id" type="string" required>
  Unique identifier of the training job
</ParamField>

## Query Parameters

<ParamField query="include_metrics" type="boolean" default="true">
  Include detailed training metrics and performance data
</ParamField>

<ParamField query="include_logs" type="boolean" default="false">
  Include training logs (for debugging)
</ParamField>

<ParamField query="metrics_limit" type="integer" default="100" min="10" max="1000">
  Maximum number of metric data points to return
</ParamField>

## Response

<ResponseField name="job_id" type="string" required>
  Unique identifier for the training job
</ResponseField>

<ResponseField name="judge_id" type="string" required>
  ID of the judge being trained
</ResponseField>

<ResponseField name="training_type" type="string" required>
  Type of training performed
</ResponseField>

<ResponseField name="status" type="string" required>
  Current job status
  - `created` - Job created but not started
  - `queued` - Job queued for execution
  - `preparing` - Preparing training data
  - `training` - Training in progress
  - `validating` - Validating results
  - `completed` - Training completed successfully
  - `failed` - Training failed
  - `cancelled` - Job was cancelled
</ResponseField>

<ResponseField name="progress" type="object" required>
  Training progress information
  
  <Expandable title="progress properties">
    <ResponseField name="current_epoch" type="integer">
      Current training epoch
    </ResponseField>
    
    <ResponseField name="total_epochs" type="integer">
      Total planned epochs
    </ResponseField>
    
    <ResponseField name="epoch_progress" type="number">
      Progress within current epoch (0.0 to 1.0)
    </ResponseField>
    
    <ResponseField name="overall_progress" type="number" required>
      Overall job progress percentage (0-100)
    </ResponseField>
    
    <ResponseField name="estimated_time_remaining" type="integer">
      Estimated minutes remaining
    </ResponseField>
    
    <ResponseField name="phase" type="string" required>
      Current training phase
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="training_config" type="object" required>
  Configuration used for training
</ResponseField>

<ResponseField name="metrics" type="object">
  Training metrics (if include_metrics=true)
  
  <Expandable title="metrics properties">
    <ResponseField name="training_loss" type="array">
      Training loss values over time
    </ResponseField>
    
    <ResponseField name="validation_loss" type="array">
      Validation loss values over time
    </ResponseField>
    
    <ResponseField name="accuracy_improvement" type="number">
      Accuracy improvement percentage
    </ResponseField>
    
    <ResponseField name="f1_score" type="number">
      F1 score on validation set
    </ResponseField>
    
    <ResponseField name="criteria_improvements" type="object">
      Per-criterion improvement metrics
    </ResponseField>
    
    <ResponseField name="convergence_info" type="object">
      Information about training convergence
      
      <Expandable title="convergence properties">
        <ResponseField name="converged" type="boolean" required>
          Whether training has converged
        </ResponseField>
        
        <ResponseField name="early_stopped" type="boolean" required>
          Whether early stopping was triggered
        </ResponseField>
        
        <ResponseField name="patience_remaining" type="integer">
          Early stopping patience remaining
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="results" type="object">
  Training results (if job completed)
  
  <Expandable title="results properties">
    <ResponseField name="new_model_version" type="string">
      Version ID of the newly trained model
    </ResponseField>
    
    <ResponseField name="performance_comparison" type="object">
      Comparison with previous version
      
      <Expandable title="comparison properties">
        <ResponseField name="accuracy_delta" type="number" required>
          Change in accuracy percentage
        </ResponseField>
        
        <ResponseField name="consistency_delta" type="number" required>
          Change in consistency score
        </ResponseField>
        
        <ResponseField name="speed_delta" type="number" required>
          Change in evaluation speed (ms)
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="validation_results" type="object">
      Results from validation testing
      
      <Expandable title="validation properties">
        <ResponseField name="test_accuracy" type="number" required>
          Accuracy on test set
        </ResponseField>
        
        <ResponseField name="test_f1_score" type="number" required>
          F1 score on test set
        </ResponseField>
        
        <ResponseField name="criteria_scores" type="object" required>
          Per-criterion validation scores
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="deployment_ready" type="boolean" required>
      Whether the new model is ready for deployment
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="error_info" type="object">
  Error information (if job failed)
  
  <Expandable title="error properties">
    <ResponseField name="error_code" type="string" required>
      Error classification code
    </ResponseField>
    
    <ResponseField name="error_message" type="string" required>
      Human-readable error description
    </ResponseField>
    
    <ResponseField name="failed_at_epoch" type="integer">
      Epoch where failure occurred
    </ResponseField>
    
    <ResponseField name="suggested_fixes" type="array">
      Suggested solutions for the error
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Job creation timestamp
</ResponseField>

<ResponseField name="started_at" type="datetime">
  Training start timestamp
</ResponseField>

<ResponseField name="completed_at" type="datetime">
  Job completion timestamp
</ResponseField>

<ResponseField name="created_by" type="string" required>
  User ID who created the job
</ResponseField>
