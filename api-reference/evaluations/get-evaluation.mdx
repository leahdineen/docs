---
title: 'Get Evaluation'
api: 'GET /evaluations/{evaluation_id}'
description: 'Retrieve a specific evaluation result'
---

## Overview

Retrieves the full details of a completed or in-progress evaluation.

## Path Parameters

<ParamField path="evaluation_id" type="string" required>
  The unique identifier of the evaluation
</ParamField>

## Response

<ResponseField name="evaluation_id" type="string" required>
  Unique evaluation identifier
</ResponseField>

<ResponseField name="content_id" type="string" required>
  ID of the evaluated content
</ResponseField>

<ResponseField name="judge_id" type="string" required>
  ID of the judge used
</ResponseField>

<ResponseField name="overall_score" type="number" required>
  Overall evaluation score (0-100)
</ResponseField>

<ResponseField name="status" type="string" required>
  Evaluation status (processing, completed, failed)
</ResponseField>

<ResponseField name="criteria_scores" type="array" required>
  Individual criteria scores
  
  <Expandable title="properties">
    <ResponseField name="criterion_id" type="string" required>
      Criterion identifier
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Criterion name
    </ResponseField>
    
    <ResponseField name="score" type="number" required>
      Score (0-100)
    </ResponseField>
    
    <ResponseField name="feedback" type="string">
      Detailed feedback
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="summary" type="string">
  Evaluation summary
</ResponseField>

<ResponseField name="strengths" type="array">
  Identified strengths
</ResponseField>

<ResponseField name="weaknesses" type="array">
  Identified weaknesses
</ResponseField>

<ResponseField name="suggestions" type="array">
  Improvement suggestions
</ResponseField>

<ResponseField name="evaluated_at" type="datetime" required>
  Evaluation completion timestamp
</ResponseField>

<ResponseField name="evaluation_duration_sec" type="number">
  Time taken to complete evaluation in seconds
</ResponseField>
