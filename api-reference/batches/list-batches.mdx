---
title: 'List Batches'
api: 'GET /batches'
description: 'Retrieve a paginated list of batch jobs'
---

## Overview

Retrieves a paginated list of batch evaluation jobs with optional filtering.

## Query Parameters

<ParamField query="limit" type="integer" default="50">
  Number of batches to return (max 100)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of batches to skip
</ParamField>

<ParamField query="status" type="string">
  Filter by status (created, evaluating, completed, failed)
</ParamField>

<ParamField query="judge_id" type="string">
  Filter by judge ID
</ParamField>

<ParamField query="since" type="datetime">
  Filter batches created after this timestamp
</ParamField>

## Response

<ResponseField name="batches" type="array" required>
  Array of batch objects
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of batches
</ResponseField>

<ResponseField name="has_more" type="boolean" required>
  Whether more batches are available
</ResponseField>

<ResponseField name="limit" type="integer">
  Number returned
</ResponseField>

<ResponseField name="offset" type="integer">
  Number skipped
</ResponseField>
