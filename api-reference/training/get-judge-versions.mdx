---
title: 'Get Judge Versions'
api: 'GET /judges/{judge_id}/versions'
description: 'Retrieve version history for a judge'
---

## Overview

Retrieves the version history for a judge, including all past versions, their performance metrics, and deployment status. This helps track judge evolution and compare different versions.

## Path Parameters

<ParamField path="judge_id" type="string" required>
  Unique identifier of the judge
</ParamField>

## Query Parameters

<ParamField query="limit" type="integer" default="25" min="1" max="100">
  Maximum number of versions to return
</ParamField>

<ParamField query="offset" type="integer" default="0" min="0">
  Number of versions to skip for pagination
</ParamField>

<ParamField query="include_metrics" type="boolean" default="true">
  Include performance metrics for each version
</ParamField>

<ParamField query="status_filter" type="string">
  Filter versions by status
  - `active` - Currently active version
  - `deployed` - Previously deployed versions
  - `draft` - Draft versions not yet deployed
  - `archived` - Archived old versions
</ParamField>

## Response

<ResponseField name="judge_id" type="string" required>
  ID of the judge
</ResponseField>

<ResponseField name="current_version" type="integer" required>
  Currently active version number
</ResponseField>

<ResponseField name="total_versions" type="integer" required>
  Total number of versions created
</ResponseField>

<ResponseField name="versions" type="array" required>
  List of judge versions
  
  <Expandable title="version properties">
    <ResponseField name="version_number" type="integer" required>
      Version number
    </ResponseField>
    
    <ResponseField name="version_id" type="string" required>
      Unique version identifier
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Version status (active, deployed, draft, archived)
    </ResponseField>
    
    <ResponseField name="created_at" type="datetime" required>
      Version creation timestamp
    </ResponseField>
    
    <ResponseField name="deployed_at" type="datetime">
      Deployment timestamp (if deployed)
    </ResponseField>
    
    <ResponseField name="archived_at" type="datetime">
      Archive timestamp (if archived)
    </ResponseField>
    
    <ResponseField name="performance_metrics" type="object">
      Performance metrics (if include_metrics=true)
      
      <Expandable title="metrics properties">
        <ResponseField name="accuracy_score" type="number" required>
          Overall accuracy score (0-100)
        </ResponseField>
        
        <ResponseField name="consistency_score" type="number" required>
          Consistency score (0-100)
        </ResponseField>
        
        <ResponseField name="average_evaluation_time" type="number" required>
          Average evaluation time in milliseconds
        </ResponseField>
        
        <ResponseField name="criteria_performance" type="object" required>
          Per-criterion performance metrics
        </ResponseField>
        
        <ResponseField name="evaluation_count" type="integer" required>
          Number of evaluations performed with this version
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="changes_summary" type="string">
      Summary of changes in this version
    </ResponseField>
    
    <ResponseField name="training_info" type="object">
      Information about training that created this version
      
      <Expandable title="training info">
        <ResponseField name="training_job_id" type="string">
          ID of training job that created this version
        </ResponseField>
        
        <ResponseField name="feedback_count_used" type="integer">
          Number of feedback entries used in training
        </ResponseField>
        
        <ResponseField name="improvements_achieved" type="array">
          List of improvements achieved
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
