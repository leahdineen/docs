---
title: 'Get Batch'
api: 'GET /batches/{batch_id}'
description: 'Retrieve batch details and progress'
---

## Overview

Retrieves detailed information about a batch job including progress and results.

## Path Parameters

<ParamField path="batch_id" type="string" required>
  The unique identifier of the batch
</ParamField>

## Response

<ResponseField name="id" type="string" required>
  Batch identifier
</ResponseField>

<ResponseField name="name" type="string" required>
  Batch name
</ResponseField>

<ResponseField name="description" type="string">
  Batch description
</ResponseField>

<ResponseField name="status" type="string" required>
  Current status (created, evaluating, completed, failed)
</ResponseField>

<ResponseField name="judge_ids" type="array" required>
  Array of judge IDs used
</ResponseField>

<ResponseField name="content_ids" type="array" required>
  Array of content IDs being evaluated
</ResponseField>

<ResponseField name="total_evaluations" type="integer" required>
  Total evaluations to perform
</ResponseField>

<ResponseField name="completed_evaluations" type="integer" required>
  Completed evaluations
</ResponseField>

<ResponseField name="failed_evaluations" type="integer" required>
  Failed evaluations
</ResponseField>

<ResponseField name="progress_percentage" type="number" required>
  Completion percentage (0-100)
</ResponseField>

<ResponseField name="results_summary" type="object">
  Summary of evaluation results
  
  <Expandable title="properties">
    <ResponseField name="average_score" type="number">
      Average score across all evaluations
    </ResponseField>
    
    <ResponseField name="score_distribution" type="object">
      Score distribution by range
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Creation timestamp
</ResponseField>

<ResponseField name="completed_at" type="datetime">
  Completion timestamp
</ResponseField>
