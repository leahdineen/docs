---
title: 'Execute Training Job'
api: 'POST /training-jobs/{job_id}/execute'
description: 'Start execution of a created training job'
---

## Overview

Starts execution of a previously created training job. Training jobs are created in a pending state and must be explicitly executed. This allows you to review job configuration and schedule execution when appropriate.

## Path Parameters

<ParamField path="job_id" type="string" required>
  Unique identifier of the training job to execute
</ParamField>

## Request Body

<ParamField body="execution_override" type="object">
  Override execution parameters (optional)
  
  <Expandable title="override properties">
    <ParamField body="priority" type="string">
      Override job priority (low, normal, high)
    </ParamField>
    
    <ParamField body="resource_allocation" type="string">
      Override resource allocation
      - `minimal` - Use minimal resources (slower but cheaper)
      - `standard` - Standard resource allocation
      - `high_performance` - High performance resources (faster but more expensive)
    </ParamField>
    
    <ParamField body="max_duration_minutes" type="integer" min="10" max="1440">
      Maximum execution time in minutes (10 min to 24 hours)
    </ParamField>
    
    <ParamField body="notification_preferences" type="object">
      Override notification settings
      
      <Expandable title="notification properties">
        <ParamField body="email_notifications" type="boolean">
          Enable email notifications
        </ParamField>
        
        <ParamField body="webhook_url" type="string">
          Override webhook URL for status updates
        </ParamField>
        
        <ParamField body="notification_frequency" type="string">
          Frequency of progress notifications
          - `milestone_only` - Only major milestones
          - `periodic` - Regular progress updates
          - `detailed` - Detailed progress updates
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="confirmation" type="object">
  Execution confirmation parameters
  
  <Expandable title="confirmation properties">
    <ParamField body="acknowledge_cost" type="boolean" default="false">
      Acknowledge training costs (required for high-cost jobs)
    </ParamField>
    
    <ParamField body="backup_confirmation" type="boolean" default="false">
      Confirm that current model backup is desired
    </ParamField>
    
    <ParamField body="risk_acceptance" type="string">
      Accept identified risks (if any were found during job creation)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="job_id" type="string" required>
  ID of the executed training job
</ResponseField>

<ResponseField name="execution_id" type="string" required>
  Unique identifier for this execution instance
</ResponseField>

<ResponseField name="status" type="string" required>
  Updated job status after execution start
  - `queued` - Job queued for execution
  - `preparing` - Preparing training environment
  - `training` - Training started
</ResponseField>

<ResponseField name="estimated_duration" type="integer" required>
  Updated estimated duration in minutes
</ResponseField>

<ResponseField name="resource_allocation" type="object" required>
  Allocated resources for training
  
  <Expandable title="resource properties">
    <ResponseField name="compute_tier" type="string" required>
      Compute tier assigned (minimal, standard, high_performance)
    </ResponseField>
    
    <ResponseField name="gpu_count" type="integer" required>
      Number of GPUs allocated
    </ResponseField>
    
    <ResponseField name="memory_gb" type="integer" required>
      Memory allocated in GB
    </ResponseField>
    
    <ResponseField name="estimated_cost" type="number" required>
      Estimated cost in credits for the training
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="execution_timeline" type="object" required>
  Expected execution timeline
  
  <Expandable title="timeline properties">
    <ResponseField name="queue_position" type="integer">
      Position in execution queue (if queued)
    </ResponseField>
    
    <ResponseField name="estimated_start_time" type="datetime" required>
      Estimated training start time
    </ResponseField>
    
    <ResponseField name="estimated_completion_time" type="datetime" required>
      Estimated completion time
    </ResponseField>
    
    <ResponseField name="milestone_schedule" type="array">
      Expected milestone timestamps
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="monitoring_info" type="object" required>
  Information for monitoring execution
  
  <Expandable title="monitoring properties">
    <ResponseField name="progress_url" type="string">
      URL to monitor real-time progress
    </ResponseField>
    
    <ResponseField name="webhook_configured" type="boolean" required>
      Whether webhook notifications are configured
    </ResponseField>
    
    <ResponseField name="log_retention_days" type="integer" required>
      How long execution logs will be retained
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="executed_at" type="datetime" required>
  Timestamp when execution was started
</ResponseField>

<ResponseField name="executed_by" type="string" required>
  User ID who executed the job
</ResponseField>

    "execution_override": {
