---
title: 'Get API Key Evaluation'
api: 'GET /evaluations/{job_id}'
description: 'Retrieve evaluation job status and results using API key'
---

## Overview

Retrieves the status, progress, and results of a batch evaluation job submitted via API key. This endpoint provides detailed information about job execution and individual item results.

## Headers

<ParamField header="X-API-Key" type="string" required>
  Organization API key for authentication
</ParamField>

## Path Parameters

<ParamField path="job_id" type="string" required>
  Unique identifier of the evaluation job
</ParamField>

## Query Parameters

<ParamField query="include_results" type="boolean" default="true">
  Include individual item evaluation results
</ParamField>

<ParamField query="results_limit" type="integer" default="100" min="1" max="1000">
  Maximum number of individual results to return
</ParamField>

<ParamField query="results_offset" type="integer" default="0" min="0">
  Number of results to skip for pagination
</ParamField>

<ParamField query="status_filter" type="string">
  Filter results by evaluation status
  - `completed` - Only completed evaluations
  - `failed` - Only failed evaluations
  - `processing` - Only processing evaluations
</ParamField>

## Response

<ResponseField name="job_id" type="string" required>
  Unique identifier for the evaluation job
</ResponseField>

<ResponseField name="status" type="string" required>
  Current job status
  - `queued` - Job queued for processing
  - `processing` - Job is being processed
  - `completed` - All items evaluated
  - `failed` - Job failed
  - `cancelled` - Job was cancelled
</ResponseField>

<ResponseField name="total_items" type="integer" required>
  Total number of items in the batch
</ResponseField>

<ResponseField name="processed_items" type="integer" required>
  Number of items processed successfully
</ResponseField>

<ResponseField name="failed_items" type="integer" required>
  Number of items that failed processing
</ResponseField>

<ResponseField name="progress_percentage" type="number" required>
  Job completion percentage (0-100)
</ResponseField>

<ResponseField name="judge_id" type="string" required>
  ID of the judge used for evaluation
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Job creation timestamp
</ResponseField>

<ResponseField name="started_at" type="datetime">
  Job processing start timestamp
</ResponseField>

<ResponseField name="completed_at" type="datetime">
  Job completion timestamp (if completed)
</ResponseField>

<ResponseField name="estimated_completion_time" type="datetime">
  Estimated completion time (if still processing)
</ResponseField>

<ResponseField name="error_message" type="string">
  Error description (if job failed)
</ResponseField>

<ResponseField name="results" type="array">
  Individual evaluation results (if include_results=true)
  
  <Expandable title="result properties">
    <ResponseField name="item_id" type="string" required>
      Internal item identifier
    </ResponseField>
    
    <ResponseField name="external_id" type="string">
      Your external reference ID
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Item evaluation status (completed, failed, processing)
    </ResponseField>
    
    <ResponseField name="overall_score" type="number">
      Overall evaluation score (0-100)
    </ResponseField>
    
    <ResponseField name="criteria_scores" type="array">
      Individual criteria scores
      <Expandable title="criteria properties">
        <ResponseField name="criterion_name" type="string" required>
          Criterion name
        </ResponseField>
        
        <ResponseField name="score" type="number" required>
          Criterion score (0-100)
        </ResponseField>
        
        <ResponseField name="feedback" type="string">
          Criterion feedback
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="summary" type="string">
      Evaluation summary
    </ResponseField>
    
    <ResponseField name="strengths" type="array">
      Identified strengths
    </ResponseField>
    
    <ResponseField name="weaknesses" type="array">
      Identified weaknesses
    </ResponseField>
    
    <ResponseField name="suggestions" type="array">
      Improvement suggestions
    </ResponseField>
    
    <ResponseField name="evaluated_at" type="datetime">
      Item evaluation timestamp
    </ResponseField>
    
    <ResponseField name="error_message" type="string">
      Error description (if item failed)
    </ResponseField>
  </Expandable>
</ResponseField>
