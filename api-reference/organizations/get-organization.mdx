---
title: 'Get Organization'
api: 'GET /organizations/me'
description: 'Retrieve current user organization information'
---

## Overview

Retrieves information about the current user's organization, including usage statistics, settings, and member information.

## Response

<ResponseField name="organization_id" type="string" required>
  Unique organization identifier
</ResponseField>

<ResponseField name="name" type="string" required>
  Organization name
</ResponseField>

<ResponseField name="description" type="string">
  Organization description
</ResponseField>

<ResponseField name="status" type="string" required>
  Organization status (active, pending, suspended)
</ResponseField>

<ResponseField name="settings" type="object" required>
  Organization configuration settings
</ResponseField>

<ResponseField name="usage_stats" type="object" required>
  Current usage statistics
  
  <Expandable title="usage properties">
    <ResponseField name="judges_count" type="integer" required>
      Number of active judges
    </ResponseField>
    
    <ResponseField name="monthly_evaluations" type="integer" required>
      Evaluations performed this month
    </ResponseField>
    
    <ResponseField name="storage_used_mb" type="number" required>
      Storage used in megabytes
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="member_count" type="integer" required>
  Number of organization members
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Organization creation timestamp
</ResponseField>
