---
title: 'Create Training Job'
api: 'POST /training/jobs'
description: 'Create a new training job for judge improvement'
---

## Overview

Creates a new training job to improve judge performance based on accumulated feedback. Training jobs process user corrections and feedback to generate improved judge models with better accuracy and consistency.

## Request Body

<ParamField body="judge_id" type="string" required>
  ID of the judge to train
</ParamField>

<ParamField body="training_type" type="string" required>
  Type of training to perform
  - `incremental` - Incremental training with recent feedback
  - `full_retrain` - Complete retraining from all feedback
  - `fine_tune` - Fine-tuning specific criteria
  - `validation` - Validation-only run (no model updates)
</ParamField>

<ParamField body="training_config" type="object" required>
  Training configuration parameters
  
  <Expandable title="training config properties">
    <ParamField body="feedback_cutoff_date" type="datetime">
      Only use feedback after this date (ISO 8601)
    </ParamField>
    
    <ParamField body="minimum_feedback_count" type="integer" default="10" min="1">
      Minimum feedback entries required to start training
    </ParamField>
    
    <ParamField body="target_criteria" type="array">
      Specific criteria to focus training on (empty = all criteria)
    </ParamField>
    
    <ParamField body="validation_split" type="number" default="0.2" min="0.1" max="0.5">
      Fraction of data to use for validation (0.1 to 0.5)
    </ParamField>
    
    <ParamField body="learning_rate" type="number" default="0.001" min="0.0001" max="0.1">
      Learning rate for training algorithm
    </ParamField>
    
    <ParamField body="batch_size" type="integer" default="32" min="8" max="128">
      Training batch size
    </ParamField>
    
    <ParamField body="max_epochs" type="integer" default="100" min="10" max="1000">
      Maximum training epochs
    </ParamField>
    
    <ParamField body="early_stopping" type="boolean" default="true">
      Enable early stopping when validation loss plateaus
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="execution_options" type="object">
  Job execution options
  
  <Expandable title="execution options">
    <ParamField body="priority" type="string" default="normal">
      Job priority level (low, normal, high)
    </ParamField>
    
    <ParamField body="auto_deploy" type="boolean" default="false">
      Automatically deploy successful training results
    </ParamField>
    
    <ParamField body="backup_current_version" type="boolean" default="true">
      Create backup of current judge version before training
    </ParamField>
    
    <ParamField body="notification_email" type="string">
      Email address for job completion notifications
    </ParamField>
    
    <ParamField body="webhook_url" type="string">
      URL to receive webhook notifications about job progress
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="job_id" type="string" required>
  Unique identifier for the training job
</ResponseField>

<ResponseField name="judge_id" type="string" required>
  ID of the judge being trained
</ResponseField>

<ResponseField name="training_type" type="string" required>
  Type of training being performed
</ResponseField>

<ResponseField name="status" type="string" required>
  Initial job status
  - `created` - Job created but not started
  - `queued` - Job queued for execution
  - `preparing` - Preparing training data
  - `training` - Training in progress
  - `validating` - Validating results
  - `completed` - Training completed successfully
  - `failed` - Training failed
  - `cancelled` - Job was cancelled
</ResponseField>

<ResponseField name="feedback_data_summary" type="object" required>
  Summary of feedback data available for training
  
  <Expandable title="feedback summary">
    <ResponseField name="total_feedback_count" type="integer" required>
      Total feedback entries available
    </ResponseField>
    
    <ResponseField name="correction_count" type="integer" required>
      Number of correction entries
    </ResponseField>
    
    <ResponseField name="rating_count" type="integer" required>
      Number of rating entries
    </ResponseField>
    
    <ResponseField name="date_range" type="object" required>
      Date range of feedback data
      <Expandable title="date range">
        <ResponseField name="earliest" type="datetime" required>
          Earliest feedback date
        </ResponseField>
        <ResponseField name="latest" type="datetime" required>
          Latest feedback date
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="estimated_duration" type="integer">
  Estimated training duration in minutes
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Job creation timestamp
</ResponseField>

<ResponseField name="created_by" type="string" required>
  User ID who created the job
</ResponseField>

## Examples

<CodeGroup>

```bash Request: Incremental Training
curl -X POST https://api.getpique.ai/training/jobs \
  -H "X-API-Key: sk_org_xxxxxxxxx" \
  -H "Content-Type: application/json" \
  -d '{
    "judge_id": "jdg_xyz789",
    "training_type": "incremental",
    "training_config": {
      "feedback_cutoff_date": "2024-01-01T00:00:00Z",
      "minimum_feedback_count": 20,

  }'
```
