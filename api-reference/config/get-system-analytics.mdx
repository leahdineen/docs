---
title: 'Get System Analytics'
api: 'GET /analytics/system'
description: 'Retrieve system-wide analytics and metrics'
---

## Overview

Retrieves system-wide analytics including overall usage statistics, performance metrics, and health indicators across all judges and organizations.

## Query Parameters

<ParamField query="time_range" type="string" default="7d">
  Time range for analytics (7d, 30d, 90d, 1y)
</ParamField>

<ParamField query="include_trends" type="boolean" default="true">
  Include trend analysis
</ParamField>

## Response

<ResponseField name="system_overview" type="object" required>
  System-wide overview metrics
  
  <Expandable title="overview properties">
    <ResponseField name="total_judges" type="integer" required>
      Total number of active judges
    </ResponseField>
    
    <ResponseField name="total_evaluations" type="integer" required>
      Total evaluations performed in time range
    </ResponseField>
    
    <ResponseField name="active_organizations" type="integer" required>
      Number of active organizations
    </ResponseField>
    
    <ResponseField name="average_system_load" type="number" required>
      Average system load percentage
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="performance_metrics" type="object" required>
  System performance metrics
  
  <Expandable title="performance properties">
    <ResponseField name="average_response_time" type="number" required>
      Average response time across all evaluations (ms)
    </ResponseField>
    
    <ResponseField name="success_rate" type="number" required>
      Evaluation success rate percentage
    </ResponseField>
    
    <ResponseField name="throughput" type="object" required>
      System throughput metrics
      
      <Expandable title="throughput properties">
        <ResponseField name="evaluations_per_hour" type="number" required>
          Average evaluations per hour
        </ResponseField>
        
        <ResponseField name="peak_concurrent_evaluations" type="integer" required>
          Peak concurrent evaluations
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="usage_breakdown" type="object" required>
  Usage breakdown by various dimensions
  
  <Expandable title="usage properties">
    <ResponseField name="by_content_type" type="object" required>
      Usage breakdown by content type
    </ResponseField>
    
    <ResponseField name="by_judge_type" type="object" required>
      Usage breakdown by judge category
    </ResponseField>
    
    <ResponseField name="by_time_of_day" type="array" required>
      Usage patterns by hour of day
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="health_status" type="object" required>
  Overall system health status
  
  <Expandable title="health properties">
    <ResponseField name="overall_status" type="string" required>
      Overall system status (healthy, warning, critical)
    </ResponseField>
    
    <ResponseField name="component_status" type="object" required>
      Status of individual system components
    </ResponseField>
    
    <ResponseField name="alerts" type="array">
      Active system alerts
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="trends" type="object">
  System trends (if include_trends=true)
</ResponseField>

<ResponseField name="generated_at" type="datetime" required>
  Analytics generation timestamp
</ResponseField>
