title: 'System Statistics'
api: 'GET /stats'
description: 'Get detailed system statistics'
---

## Overview

Returns comprehensive statistics about judges, content, evaluations, and storage usage.

## Response

<ResponseField name="judges" type="object" required>
  Judge statistics
  
  <Expandable title="properties">
    <ResponseField name="total" type="integer" required>
      Total number of judges
    </ResponseField>
    
    <ResponseField name="by_type" type="object" required>
      Judges grouped by content type
      - `text` - Number of text judges
      - `video` - Number of video judges
      - `audio` - Number of audio judges
    </ResponseField>
    
    <ResponseField name="by_status" type="object" required>
      Judges grouped by status
      - `ready` - Judges ready for use
      - `learning` - Judges in learning mode
      - `training` - Judges being trained
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="content" type="object" required>
  Content statistics
  
  <Expandable title="properties">
    <ResponseField name="items" type="integer" required>
      Total content items
    </ResponseField>
    
    <ResponseField name="evaluations" type="integer" required>
      Total evaluations performed
    </ResponseField>
    
    <ResponseField name="batches" type="integer" required>
      Total batch jobs
    </ResponseField>
    
    <ResponseField name="feedback_items" type="integer" required>
      Total feedback entries
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="storage" type="object" required>
  Storage statistics
  
  <Expandable title="properties">
    <ResponseField name="total_size" type="string" required>
      Total storage used (e.g., "125.3MB")
    </ResponseField>
    
    <ResponseField name="files_by_type" type="object" required>
      File count by type
      - `text` - Number of text files
      - `video` - Number of video files
      - `audio` - Number of audio files
      - `image` - Number of image files
    </ResponseField>
  </Expandable>
</ResponseField>
