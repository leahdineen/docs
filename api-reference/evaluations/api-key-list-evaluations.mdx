---
title: 'List API Key Evaluations'
api: 'GET /evaluations'
description: 'List evaluation jobs for your organization'
---

## Overview

Retrieves a list of evaluation jobs submitted by your organization using API key authentication. Supports filtering, sorting, and pagination for efficient job management.

## Headers

<ParamField header="X-API-Key" type="string" required>
  Organization API key for authentication
</ParamField>

## Query Parameters

<ParamField query="status" type="string">
  Filter jobs by status
  - `queued` - Jobs waiting to be processed
  - `processing` - Jobs currently being processed
  - `completed` - Successfully completed jobs
  - `failed` - Failed jobs
  - `cancelled` - Cancelled jobs
</ParamField>

<ParamField query="judge_id" type="string">
  Filter jobs by judge ID
</ParamField>

<ParamField query="created_after" type="datetime">
  Filter jobs created after this timestamp (ISO 8601)
</ParamField>

<ParamField query="created_before" type="datetime">
  Filter jobs created before this timestamp (ISO 8601)
</ParamField>

<ParamField query="limit" type="integer" default="25" min="1" max="100">
  Maximum number of jobs to return
</ParamField>

<ParamField query="offset" type="integer" default="0" min="0">
  Number of jobs to skip for pagination
</ParamField>

<ParamField query="sort_by" type="string" default="created_at">
  Field to sort results by
  - `created_at` - Creation timestamp
  - `completed_at` - Completion timestamp
  - `progress_percentage` - Job progress
  - `total_items` - Number of items in job
</ParamField>

<ParamField query="sort_order" type="string" default="desc">
  Sort order
  - `asc` - Ascending order
  - `desc` - Descending order
</ParamField>

## Response

<ResponseField name="jobs" type="array" required>
  List of evaluation jobs
  
  <Expandable title="job properties">
    <ResponseField name="job_id" type="string" required>
      Unique job identifier
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Job status (queued, processing, completed, failed, cancelled)
    </ResponseField>
    
    <ResponseField name="total_items" type="integer" required>
      Total number of items in the job
    </ResponseField>
    
    <ResponseField name="processed_items" type="integer" required>
      Number of items processed
    </ResponseField>
    
    <ResponseField name="failed_items" type="integer" required>
      Number of items that failed
    </ResponseField>
    
    <ResponseField name="progress_percentage" type="number" required>
      Job completion percentage (0-100)
    </ResponseField>
    
    <ResponseField name="judge_id" type="string" required>
      ID of the judge used
    </ResponseField>
    
    <ResponseField name="judge_name" type="string">
      Name of the judge used
    </ResponseField>
    
    <ResponseField name="created_at" type="datetime" required>
      Job creation timestamp
    </ResponseField>
    
    <ResponseField name="started_at" type="datetime">
      Processing start timestamp
    </ResponseField>
    
    <ResponseField name="completed_at" type="datetime">
      Job completion timestamp (if completed)
    </ResponseField>
    
    <ResponseField name="estimated_completion_time" type="datetime">
      Estimated completion time (if processing)
    </ResponseField>
    
    <ResponseField name="priority" type="string">
      Job priority level (low, normal, high)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object" required>
  Pagination information
  
  <Expandable title="pagination properties">
    <ResponseField name="total_count" type="integer" required>
      Total number of jobs matching filters
    </ResponseField>
    
    <ResponseField name="limit" type="integer" required>
      Current limit parameter
    </ResponseField>
    
    <ResponseField name="offset" type="integer" required>
      Current offset parameter
    </ResponseField>
    
    <ResponseField name="has_next" type="boolean" required>
      Whether more results are available
    </ResponseField>
    
    <ResponseField name="has_previous" type="boolean" required>
      Whether previous results exist
    </ResponseField>
  </Expandable>
</ResponseField>
