---
title: 'Create Organization'
api: 'POST /organizations'
description: 'Create a new organization'
---

## Overview

Creates a new organization with administrative access controls and resource allocation settings. Organizations provide isolated environments for teams and manage access to judges, content, and evaluations.

## Request Body

<ParamField body="name" type="string" required maxLength="100">
  Organization name
</ParamField>

<ParamField body="description" type="string" maxLength="500">
  Organization description
</ParamField>

<ParamField body="settings" type="object">
  Organization configuration settings
  
  <Expandable title="settings properties">
    <ParamField body="max_judges" type="integer" default="10" min="1" max="100">
      Maximum number of judges allowed
    </ParamField>
    
    <ParamField body="max_monthly_evaluations" type="integer" default="10000">
      Monthly evaluation limit
    </ParamField>
    
    <ParamField body="data_retention_days" type="integer" default="365">
      Data retention period in days
    </ParamField>
    
    <ParamField body="allowed_content_types" type="array" default="[\"text\", \"image\", \"video\", \"audio\"]">
      Allowed content types for evaluation
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="organization_id" type="string" required>
  Unique organization identifier
</ResponseField>

<ResponseField name="name" type="string" required>
  Organization name
</ResponseField>

<ResponseField name="description" type="string">
  Organization description
</ResponseField>

<ResponseField name="status" type="string" required>
  Organization status (active, pending, suspended)
</ResponseField>

<ResponseField name="settings" type="object" required>
  Organization settings
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Creation timestamp
</ResponseField>

<ResponseField name="admin_user_id" type="string" required>
  ID of the user who created and administers this organization
</ResponseField>

## Examples

```bash Request

  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Acme Content Team",
    "description": "Content evaluation team for marketing materials",
    "settings": {
      "max_judges": 15,
      "max_monthly_evaluations": 25000,
      "data_retention_days": 730
    }
  }'
```
