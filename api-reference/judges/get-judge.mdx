---
title: 'Get Judge'
api: 'GET /judges/{judge_id}'
description: 'Retrieve details of a specific judge'
---

## Overview

Retrieves complete details of a judge including its criteria and analytics.

## Path Parameters

<ParamField path="judge_id" type="string" required>
  The unique identifier of the judge
</ParamField>

## Response

<ResponseField name="id" type="string" required>
  Unique judge identifier
</ResponseField>

<ResponseField name="name" type="string" required>
  Judge name
</ResponseField>

<ResponseField name="description" type="string" required>
  Judge description
</ResponseField>

<ResponseField name="content_type" type="string" required>
  Content type (text, video, audio, image)
</ResponseField>

<ResponseField name="status" type="string" required>
  Judge status (ready, learning, training)
</ResponseField>

<ResponseField name="training_progress" type="integer" required>
  Training progress (0-100)
</ResponseField>

<ResponseField name="criteria" type="array" required>
  List of evaluation criteria
  
  <Expandable title="criterion properties">
    <ResponseField name="id" type="string" required>
      Criterion ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Criterion name
    </ResponseField>
    
    <ResponseField name="description" type="string" required>
      Criterion description
    </ResponseField>
    
    <ResponseField name="version_number" type="integer" required>
      Version number
    </ResponseField>
    
    <ResponseField name="is_active" type="boolean" required>
      Active status
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="analytics" type="object" required>
  Judge usage analytics
  
  <Expandable title="analytics properties">
    <ResponseField name="total_evaluations" type="integer" required>
      Total number of evaluations performed
    </ResponseField>
    
    <ResponseField name="average_score" type="number" required>
      Average evaluation score
    </ResponseField>
    
    <ResponseField name="last_evaluated_at" type="datetime">
      Last evaluation timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created_at" type="datetime" required>
  Creation timestamp
</ResponseField>

<ResponseField name="updated_at" type="datetime" required>
  Last update timestamp
</ResponseField>

<ResponseField name="version_number" type="integer" required>
  Current version
</ResponseField>

<ResponseField name="is_active" type="boolean" required>
  Active status
</ResponseField>
