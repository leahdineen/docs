---
title: 'List Content'
api: 'GET /content'
description: 'Retrieve a paginated list of content items'
---

## Overview

Retrieves a paginated list of uploaded content items with optional filtering.

## Query Parameters

<ParamField query="limit" type="integer" default="50">
  Number of content items to return (max 100)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of content items to skip
</ParamField>

<ParamField query="file_type" type="string">
  Filter by file type (text, video, audio, image)
</ParamField>

<ParamField query="status" type="string">
  Filter by status (uploaded, processing, ready, error)
</ParamField>

<ParamField query="search" type="string">
  Search in content names and descriptions
</ParamField>

<ParamField query="since" type="datetime">
  Filter content uploaded after this timestamp
</ParamField>

## Response

<ResponseField name="content" type="array" required>
  Array of content objects
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of content items
</ResponseField>

<ResponseField name="has_more" type="boolean" required>
  Whether more content is available
</ResponseField>

<ResponseField name="limit" type="integer">
  Number returned
</ResponseField>

<ResponseField name="offset" type="integer">
  Number skipped
</ResponseField>
