---
title: 'Get Judge Analytics'
api: 'GET /analytics/judges/{judge_id}'
description: 'Retrieve analytics and performance metrics for a judge'
---

## Overview

Retrieves comprehensive analytics and performance metrics for a specific judge, including accuracy trends, usage statistics, and improvement recommendations.

## Path Parameters

<ParamField path="judge_id" type="string" required>
  Unique identifier of the judge
</ParamField>

## Query Parameters

<ParamField query="time_range" type="string" default="30d">
  Time range for analytics
  - `7d` - Last 7 days
  - `30d` - Last 30 days
  - `90d` - Last 90 days
  - `1y` - Last year
</ParamField>

<ParamField query="include_trends" type="boolean" default="true">
  Include performance trend analysis
</ParamField>

## Response

<ResponseField name="judge_id" type="string" required>
  ID of the judge
</ResponseField>

<ResponseField name="judge_name" type="string" required>
  Name of the judge
</ResponseField>

<ResponseField name="analytics_period" type="object" required>
  Time period for analytics data
</ResponseField>

<ResponseField name="performance_summary" type="object" required>
  Overall performance summary
  
  <Expandable title="performance properties">
    <ResponseField name="total_evaluations" type="integer" required>
      Total evaluations performed
    </ResponseField>
    
    <ResponseField name="average_accuracy" type="number" required>
      Average accuracy score
    </ResponseField>
    
    <ResponseField name="consistency_score" type="number" required>
      Consistency score across evaluations
    </ResponseField>
    
    <ResponseField name="average_response_time" type="number" required>
      Average response time in milliseconds
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="usage_statistics" type="object" required>
  Usage statistics
  
  <Expandable title="usage properties">
    <ResponseField name="evaluations_per_day" type="array" required>
      Daily evaluation counts
    </ResponseField>
    
    <ResponseField name="peak_usage_hours" type="array" required>
      Hours with highest usage
    </ResponseField>
    
    <ResponseField name="content_type_breakdown" type="object" required>
      Breakdown by content type evaluated
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback_summary" type="object" required>
  Feedback and improvement metrics
  
  <Expandable title="feedback properties">
    <ResponseField name="total_feedback_count" type="integer" required>
      Total feedback received
    </ResponseField>
    
    <ResponseField name="approval_rate" type="number" required>
      Percentage of evaluations approved
    </ResponseField>
    
    <ResponseField name="common_issues" type="array" required>
      Most common issues identified in feedback
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="trends" type="object">
  Performance trends (if include_trends=true)
  
  <Expandable title="trend properties">
    <ResponseField name="accuracy_trend" type="string" required>
      Accuracy trend direction (improving, stable, declining)
    </ResponseField>
    
    <ResponseField name="speed_trend" type="string" required>
      Response speed trend (improving, stable, declining)
    </ResponseField>
    
    <ResponseField name="usage_trend" type="string" required>
      Usage trend (increasing, stable, decreasing)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="recommendations" type="array">
  Improvement recommendations based on analytics
</ResponseField>
