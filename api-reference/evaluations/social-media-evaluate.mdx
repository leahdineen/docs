---
title: 'Social Media Evaluation'
api: 'POST /evaluations/social-media'

---

## Overview

Evaluates social media content against platform-specific guidelines, content policies, and quality standards. This endpoint handles various social media formats and provides specialized analysis for posts, comments, stories, and other social content.

## Request Body

<ParamField body="content" type="string" required maxLength="10000">
  The social media content to evaluate (text, caption, or description)
</ParamField>

<ParamField body="platform" type="string" required>
  Social media platform type
  - `twitter` - Twitter/X posts and replies
  - `facebook` - Facebook posts and comments
  - `instagram` - Instagram posts and stories
  - `linkedin` - LinkedIn posts and articles
  - `tiktok` - TikTok video descriptions
  - `youtube` - YouTube video descriptions and comments
  - `reddit` - Reddit posts and comments
</ParamField>

<ParamField body="content_type" type="string" required>
  Type of social media content
  - `post` - Main content post
  - `comment` - Comment or reply
  - `story` - Temporary story content
  - `bio` - Profile biography
  - `caption` - Media caption
</ParamField>

<ParamField body="judge_id" type="string" required>
  ID of the judge to use for evaluation
</ParamField>

<ParamField body="metadata" type="object">
  Additional content metadata
  
  <Expandable title="metadata properties">
    <ParamField body="hashtags" type="array">
      List of hashtags in the content
    </ParamField>
    
    <ParamField body="mentions" type="array">
      List of user mentions in the content
    </ParamField>
    
    <ParamField body="media_count" type="integer">
      Number of media attachments
    </ParamField>
    
    <ParamField body="target_audience" type="string">
      Intended audience (general, teen, adult, professional)
    </ParamField>
    
    <ParamField body="language" type="string">
      Content language code (e.g., "en", "es", "fr")
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="evaluation_options" type="object">
  Evaluation configuration options
  
  <Expandable title="options properties">
    <ParamField body="check_toxicity" type="boolean" default="true">
      Enable toxicity detection
    </ParamField>
    
    <ParamField body="check_spam" type="boolean" default="true">
      Enable spam detection
    </ParamField>
    
    <ParamField body="check_compliance" type="boolean" default="true">
      Enable platform compliance checking
    </ParamField>
    
    <ParamField body="sentiment_analysis" type="boolean" default="false">
      Enable sentiment analysis
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="evaluation_id" type="string" required>
  Unique identifier for the evaluation
</ResponseField>

<ResponseField name="platform" type="string" required>
  Social media platform
</ResponseField>

<ResponseField name="content_type" type="string" required>
  Type of social content evaluated
</ResponseField>

<ResponseField name="overall_score" type="number" required>
  Overall evaluation score (0-100)
</ResponseField>

<ResponseField name="compliance_status" type="string" required>
  Platform compliance status
  - `compliant` - Meets all platform guidelines
  - `warning` - Minor issues detected
  - `violation` - Policy violations found
  - `banned` - Content would be banned/removed
</ResponseField>

<ResponseField name="criteria_scores" type="array" required>
  Individual criteria evaluation results
  
  <Expandable title="criteria properties">
    <ResponseField name="criterion_name" type="string" required>
      Name of the evaluated criterion
    </ResponseField>
    
    <ResponseField name="score" type="number" required>
      Score for this criterion (0-100)
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Criterion status (pass, warning, fail)
    </ResponseField>
    
    <ResponseField name="feedback" type="string">
      Detailed feedback for this criterion
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="risk_factors" type="array">
  Identified risk factors
  
  <Expandable title="risk factor properties">
    <ResponseField name="type" type="string" required>
      Type of risk (toxicity, spam, misinformation, etc.)
    </ResponseField>
    
    <ResponseField name="severity" type="string" required>
      Risk severity (low, medium, high, critical)
    </ResponseField>
    
    <ResponseField name="description" type="string" required>
      Description of the risk
    </ResponseField>
    
    <ResponseField name="confidence" type="number" required>
      Detection confidence (0.0 to 1.0)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="recommendations" type="array">
  Content improvement recommendations
</ResponseField>

<ResponseField name="sentiment" type="object">
  Sentiment analysis results (if enabled)
  
  <Expandable title="sentiment properties">
    <ResponseField name="polarity" type="string">
      Sentiment polarity (positive, negative, neutral)
    </ResponseField>
    
    <ResponseField name="confidence" type="number">
      Sentiment confidence score (0.0 to 1.0)
    </ResponseField>
    
    <ResponseField name="emotions" type="array">
      Detected emotions with scores
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="evaluated_at" type="datetime" required>
  Evaluation timestamp
</ResponseField>
