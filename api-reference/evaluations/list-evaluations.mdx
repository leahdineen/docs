---
title: 'List Evaluations'
api: 'GET /evaluations'
description: 'Retrieve a paginated list of evaluations'
---

## Overview

Retrieves a paginated list of evaluations with optional filtering.

## Query Parameters

<ParamField query="limit" type="integer" default="50">
  Number of evaluations to return (max 100)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of evaluations to skip
</ParamField>

<ParamField query="judge_id" type="string">
  Filter by judge ID
</ParamField>

<ParamField query="content_id" type="string">
  Filter by content ID
</ParamField>

<ParamField query="status" type="string">
  Filter by status (processing, completed, failed)
</ParamField>

<ParamField query="since" type="datetime">
  Filter evaluations after this timestamp
</ParamField>

## Response

<ResponseField name="evaluations" type="array" required>
  Array of evaluation objects
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of evaluations
</ResponseField>

<ResponseField name="has_more" type="boolean" required>
  Whether more evaluations are available
</ResponseField>

<ResponseField name="limit" type="integer">
  Number returned
</ResponseField>

<ResponseField name="offset" type="integer">
  Number skipped
</ResponseField>
