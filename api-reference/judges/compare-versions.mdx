---
title: 'Compare Judge Versions'
api: 'GET /judges/{judge_id}/versions/compare'
description: 'Compare two versions of a judge'
---

## Overview

Compares two versions of a judge to show differences in criteria and configuration.

## Path Parameters

<ParamField path="judge_id" type="string" required>
  The unique identifier of the judge
</ParamField>

## Query Parameters

<ParamField query="from_version" type="integer" required>
  Source version number for comparison
</ParamField>

<ParamField query="to_version" type="integer" required>
  Target version number for comparison
</ParamField>

## Response

<ResponseField name="from_version" type="integer" required>
  Source version number
</ResponseField>

<ResponseField name="to_version" type="integer" required>
  Target version number
</ResponseField>

<ResponseField name="differences" type="object" required>
  Differences between versions
  
  <Expandable title="differences properties">
    <ResponseField name="name_changed" type="boolean" required>
      Whether the name changed
    </ResponseField>
    
    <ResponseField name="description_changed" type="boolean" required>
      Whether the description changed
    </ResponseField>
    
    <ResponseField name="criteria_added" type="array" required>
      Criteria added in the new version
    </ResponseField>
    
    <ResponseField name="criteria_removed" type="array" required>
      Criteria removed from the old version
    </ResponseField>
    
    <ResponseField name="criteria_modified" type="array" required>
      Criteria that were modified
    </ResponseField>
  </Expandable>
</ResponseField>
